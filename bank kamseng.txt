    const gridEl = document.getElementById("grid");
    const statusEl = document.getElementById("status");
    const searchInput = document.getElementById("searchInput");

    function renderCaptions(filter = "") {
      const q = filter.toLowerCase().trim();
      gridEl.innerHTML = "";

      const filtered = CAPTIONS.filter(item => {
        if (!q) return true;
        const haystack = (item.id + " " + item.product + " " + item.tags + " " + item.text).toLowerCase();
        return haystack.includes(q);
      });

      if (filtered.length === 0) {
        gridEl.innerHTML = "<div style='font-size:13px;color:#9ca3af;'>Tidak ada caption yang cocok dengan kata kunci tersebut.</div>";
        statusEl.textContent = "";
        return;
      }

      filtered.forEach(item => {
        const card = document.createElement("div");
        card.className = "card";

        card.innerHTML = `
          <div class="pill">Klik untuk copy</div>
          <div class="product-id">No. ${item.id}</div>
          <div class="product-name">${item.product}</div>
          <div class="tags">${item.tags}</div>
          <div class="caption-text">${item.text.replace(/\\n/g, "<br>")}</div>
        `;

        card.addEventListener("click", () => handleCopy(item.text, card, item));

        gridEl.appendChild(card);
      });

      statusEl.textContent = `${filtered.length} caption ditemukan.`;
    }

    function handleCopy(text, cardEl, item) {
      const cleanText = text.trim();
      navigator.clipboard.writeText(cleanText).then(() => {
        cardEl.classList.add("copied");
        statusEl.innerHTML = `Caption <span class="highlight">No. ${item.id} ‚Äì ${item.product}</span> sudah di-copy. Tinggal paste di IG/TikTok/Shopee.`;
        setTimeout(() => {
          cardEl.classList.remove("copied");
        }, 1200);
      }).catch(err => {
        console.error("Gagal copy:", err);
        alert("Browser tidak mengizinkan copy otomatis. Coba pakai Chrome / Edge terbaru atau jalankan file ini dari hosting (https).");
      });
    }

    // Event: search
    searchInput.addEventListener("input", () => {
      renderCaptions(searchInput.value);
